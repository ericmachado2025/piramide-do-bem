### 1.3 — Unificar user_id → users_id em todas as tabelas

O problema: `user_id` aponta para `auth.users.id`, `users_id` aponta para `public.users.id`.
O correto é usar APENAS `users_id` com FK para `public.users`, que por sua vez tem `auth_id` → `auth.users`.

```sql
-- Verificar se students.users_id está sendo populado
-- Se não estiver, popular a partir de user_id via join com public.users.auth_id
UPDATE public.students s
SET users_id = u.id
FROM public.users u
WHERE u.auth_id = s.user_id
  AND s.users_id IS NULL;

UPDATE public.teachers t
SET users_id = u.id
FROM public.users u
WHERE u.auth_id = t.user_id
  AND t.users_id IS NULL;

UPDATE public.parents p
SET users_id = u.id
FROM public.users u
WHERE u.auth_id = p.user_id
  AND p.users_id IS NULL;

UPDATE public.sponsors sp
SET users_id = u.id
FROM public.users u
WHERE u.auth_id = sp.user_id
  AND sp.users_id IS NULL;
```

Criar `src/lib/userProfile.ts`:

```typescript
import { supabase } from './supabase'

// Busca dados completos do usuário logado (users + perfil)
export async function getFullProfile(authUserId: string) {
  const { data: user } = await supabase
    .from('users')
    .select('*')
    .eq('auth_id', authUserId)
    .maybeSingle()
  return user
}

// Busca aluno com dados do usuário via join
export async function getStudentWithUser(authUserId: string) {
  const { data } = await supabase
    .from('students')
    .select(`
      *,
      user:users!students_users_id_fkey(name, email, phone, whatsapp, whatsapp_visibility, verified_email, verified_phone)
    `)
    .eq('user_id', authUserId)
    .maybeSingle()
  return data
}
```
